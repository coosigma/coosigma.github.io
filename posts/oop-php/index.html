OOP in PHP

Basic Syntax

    class FOO extends POO implements IOO1, IOO2, IOO3 {
        public $name;
        function __construct($name) {
            parent::__construct();
            $this->$name = $name;
        }

        public toString() : ?string {
            return $this->name;
        }

        function __destruct() {
            echo 'Destroying: ', $this->name, PHP_EOL;
        }
    }
