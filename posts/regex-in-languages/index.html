Regex in Languages

Perl

$foo =~ /$pattern/

Match groups

$&, $1, ...

Replace

$string =~ s/pattern/replacement/g;

Reference

https://perldoc.perl.org/perlre.html

Go

import ("regexp")
match, _ := regexp.MatchString("pattern", "string")

Capture

var r = regexp.MustCompile(`pattern`)
fmt.Println(r.FindStringSubmatch("string"))
// => [group0, group1]
fmt.Println(r.MatchAllString("string1, string2", -1))
// => [string1, string2]

Replace

newString := r.RelaceAllString("oldString", "replacement")

Reference

https://golang.org/pkg/regexp/
https://gobyexample.com/regular-expressions

Ruby

'string' =~ /pattern/
/pattern/ =~ 'string'
m = 'string'.match(/pattern/)

Match groups

(Same as Perl)
require 'pp'
pp m
(m[0] is the "str" if it matched, otherwise it's nil)

Replace

'string'.gsub(/(o+)/, '\1\1\1') #=> "foooooo"

Reference

https://www.rubyguides.com/2015/06/ruby-regex/

PHP

preg_match("/pattern/", $str, $matches)

Match groups

print_f($matches)
($matches[0] is the $str if it matched, otherwise it's empty)

Replace

$newString = preg_replace("/pattern/"), $replacement, $str);

Reference

https://www.w3schools.com/php/php_regex.asp

JavaScript

/pattern/.test("string")

Match groups

var Array = str.match(/pattern/)
console.log(Array)
(Array[0] is the str if it matched, otherwise it's null)

Replace

str.replace(/pattern/, "$&")

Reference

https://developer.mozilla.org/zh-CN/docs/Web/JavaScript/Guide/Regular_Expressions

Python

import re
m = re.search(r'pattern', text)

Match groups

import pprint
pprint(m)
(m.group(0)/m[0] is the text if it matched, otherwise it's None)

Replace

newString = re.sub(r'pattern', replacement, oldString)

Reference

https://docs.python.org/3/library/re.html

C#

using System.Text.RegularExpressions;
MatchCollection mc = Regex.Matches(text, @'pattern')

Match groups

foreach (Match m in mc) {
  Console.lWriteLine(m);
}

Replace

Regex rgx = new Regex(@'pattern');
string newString = rgx.Replace(oldString, relacement);

Reference

https://www.c-sharpcorner.com/article/c-sharp-regex-examples/

Java

import java.util.regex.*;
String pattern = "pattern";
boolean isMatch = Pattern.matches(pattern, content);

Match group

import java.util.regex.Matcher;
import java.util.regex.Pattern;
Pattern r  = Pattern.compile("pattern");
Matcher m = r.matcher("string");
if (m.find()) {
  System.out.println("Found value: ", m.group(0));
} else {
  System.out.println("No Match");
}

Replace

Pattern p = Pattern.compile("pattern");
Matcher m = p.matcher("string");
String newString = m.replaceFirst("replacement");
newString = m.replaceAll("relacement");

Reference

https://www.runoob.com/java/java-regular-expressions.html
